safe_clone() {
    local url=""
    local dir=""
    local args=()
    # --- Парсим аргументы ---
    while [[ $# -gt 0 ]]; do
        case "$1" in
            --*)  # флаги git clone
                args+=("$1")
                ;;
            *)    # url и каталог назначения
                if [[ -z "$url" ]]; then
                    url="$1"
                elif [[ -z "$dir" ]]; then
                    dir="$1"
                else
                    args+=("$1")
                fi
                ;;
        esac
        shift
    done
    # echo "src: $url"
    # echo "dst: $dir"
    # echo "args: ${args[@]}"
    # --- Проверка URL ---
    if [[ -z "$url" ]]; then
        echo "safe_clone: error: URL not specified" >&2
        return 1
    fi
    # --- Если каталог не указан — берём из URL ---
    if [[ -z "$dir" ]]; then
        dir="$(basename "$url" .git)"
    fi
    # --- Проверка существующего репозитория ---
    if [[ -d "$dir/.git" ]]; then
        if git -C "$dir" fsck --no-progress --no-dangling >/dev/null 2>&1; then
            return 0
        fi
    fi
    # --- Клонирование ---
    rm -rf "$dir"
    git clone "$url" "$dir" "${args[@]}"
}

import requests
from tqdm import tqdm # sudo pip install tqdm
from io import BytesIO

def open_url(url):
    user_agent = 'Mozilla/5.0 (X11; Linux x86_64; rv:109.0) Gecko/20100101 Firefox/119.0'
    headers = {'User-Agent': user_agent}

    # stream=True — чтобы читать по кускам
    with requests.get(url, headers=headers, stream=True, timeout=300) as r:
        r.raise_for_status()

        total = int(r.headers.get('Content-Length', 0))
        buf = BytesIO()

        chunk_size = 8192
        with tqdm(
            total=total,
            unit='B',
            unit_scale=True,
            unit_divisor=1024,
            desc=f"Downloading {url.split('/')[-1]}"
        ) as bar:
            for chunk in r.iter_content(chunk_size=chunk_size):
                if chunk:
                    buf.write(chunk)
                    bar.update(len(chunk))

        buf.seek(0)
        return buf
