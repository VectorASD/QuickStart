safe_clone() {
    local url=""
    local dir=""
    local args=()

    # --- Парсим, учитывая двухтокеновые аргументы (see: git clone --help) ---
    local flags_with_arg=(
        --reference
        --reference-if-able
        -o
        --origin
        -b
        --branch
        -u
        --upload-pack
        -c
        --config
        --depth
        -j
        --jobs
    )
    while [[ $# -gt 0 ]]; do
        case "$1" in
            -*)  # флаги git clone
                args+=("$1")
                if [[ " ${flags_with_arg[*]} " == *" $1 "* ]]; then
                    args+=("$2")
                    shift
                fi
                ;;
            *)    # url и каталог назначения
                if [[ -z "$url" ]]; then
                    url="$1"
                elif [[ -z "$dir" ]]; then
                    dir="$1"
                else
                    args+=("$1")
                fi
                ;;
        esac
        shift
    done
    # echo "src: $url"
    # echo "dst: $dir"
    # echo "args: ${args[@]}"

    # --- Проверка URL ---
    if [[ -z "$url" ]]; then
        echo "safe_clone: error: URL not specified" >&2
        return 1
    fi

    # --- Если каталог не указан — берём из URL ---
    if [[ -z "$dir" ]]; then
        dir="$(basename "$url" .git)"
    fi

    # --- Проверка существующего репозитория ---
    if [[ -d "$dir/.git" ]]; then
        if git -C "$dir" fsck --no-progress --no-dangling >/dev/null 2>&1; then
            return 0
        fi
    fi

    # --- Клонирование ---
    rm -rf "$dir"
    git clone "$url" "$dir" "${args[@]}"

    # --- Локализация objects ---
    mv $dir/.git/objects $dir-objects
    ln -s $(realpath $dir-objects) $dir/.git/objects
}

def cdiv(div, rem):
    return (div + rem - 1) // rem

def calculate_list_of_bit_sizes():
    GGML_TYPE_BITS = {
        "F32": 32, "F16": 16,
        "Q4_0": 4, "Q4_1": 4, "Q5_0": 5, "Q5_1": 5, "Q8_0": 8, "Q8_1": 8,
        "Q2_K": 2, "Q3_K": 3, "Q4_K": 4, "Q5_K": 5, "Q6_K": 6, "Q8_K": 8,
        "IQ2_XXS": 2, "IQ2_XS": 2, "IQ3_XXS": 3, "IQ1_S": 1, "IQ4_NL": 4, "IQ3_S": 3, "IQ2_S": 2, "IQ4_XS": 4,
        "I8": 8, "I16": 16, "I32": 32, "I64": 64, "F64": 64,
        "IQ1_M": 1, "MXFP4": 4, "NVFP4": 4
    }
    LIST_OF_BITS = [None] * llama_cpp.GGML_TYPE_COUNT
    for k, v in llama_cpp.__dict__.items():
        if k.startswith("GGML_TYPE_") and k != "GGML_TYPE_COUNT":
            LIST_OF_BITS[v] = GGML_TYPE_BITS[k[len("GGML_TYPE_"):]]
    return LIST_OF_BITS

def get_KV_cache_size():
    n_ctx    = llm.input_ids.shape[0]   # llm.context_params.n_ctx
    metadata = llm._model.metadata()

    n_layer   = int(metadata["deepseek2.block_count"])
    n_kv_head = int(metadata["deepseek2.attention.head_count_kv"])
    key_len   = int(metadata["deepseek2.attention.key_length"])
    value_len = int(metadata["deepseek2.attention.value_length"])

    LIST_OF_BITS = calculate_list_of_bit_sizes()
    K_size = cdiv(n_layer * n_ctx * n_kv_head * key_len   * LIST_OF_BITS[llm.context_params.type_k], 8)
    V_size = cdiv(n_layer * n_ctx * n_kv_head * value_len * LIST_OF_BITS[llm.context_params.type_v], 8)

    print(f"K-cache size: {K_size / 1024 ** 3:.3f} Gb.")   # 1.266 Gb.
    print(f"V-cache size: {V_size / 1024 ** 3:.3f} Gb.")   # 0.844 Gb.
    print(f"TOTAL-cache size: {(K_size + V_size) / 1024 ** 3:.3f} Gb.")   # 2.109 Gb.



class EndDetector:
    def __init__(self, word):
        self.tokens = llm.tokenize(word.encode("utf-8"), add_bos=False)

    def match(self, history):
        tokens = self.tokens
        size = len(tokens)
        for i in range(size, 0, -1):
            if history[-i:] == tokens[:i]:
                return i, i == size
        return 0, False

    def complete(self, history):
        count, _ = self.match(history)
        history.extend(self.tokens[count:])

class Utf8Chainer:
    def __init__(self):
        self.buffer = bytearray()
        self.need = 0

    def __call__(self, bytes):
        result = bytearray()
        buffer = self.buffer

        for byte in bytes:
            if self.need:
                buffer.append(byte)
                if byte >> 6 == 0b10: # 10xxxxxx
                    self.need -= 1
                    if not self.need:
                        result += buffer
                        buffer.clear()
                else: # error
                    self.need = 0
                    result += buffer
                    buffer.clear()
            elif byte >> 7 == 0b0: # 0xxxxxxx
                result.append(byte)
            elif byte >> 5 == 0b110: # 110xxxxx
                buffer.append(byte)
                self.need = 1
            elif byte >> 4 == 0b1110: # 1110xxxx
                buffer.append(byte)
                self.need = 2
            elif byte >> 3 == 0b11110: # 11110xxx
                buffer.append(byte)
                self.need = 3
            else: # error
                result.append(byte)
        return result.decode("utf-8", errors="replace")

history = []

def core(message="", reset=False):
    if reset:
        llm.reset()
        history.clear()
    if not message:
        return
    ender     = EndDetector("\nUser:")
    chainer   = Utf8Chainer()
    ender.complete(history)
    history.extend(llm.tokenize(f" {message} \nAssistant:".encode("utf-8"), add_bos=False))
    start     = True
    prev_skip = 0
    bos_token_id: int = llm.token_bos()
    for token in llm.generate(system_prompt + history, reset=False):
        if token == bos_token_id:
            break
        history.append(token)
        skip, is_end = ender.match(history)
        count = prev_skip + 1 - skip   # сколько выпало токенов из последовательности "\nUser:" 
        if is_end:
            break
        for i in range(-count, 0):
            token = history[i-skip]  # действительные смещения выпавших токенов
            text = llm.detokenize((token,))
            if start:
                text = text.lstrip(b' ')
                start = False
            yield chainer(text)
        prev_skip = skip
    yield '\n'

def REPL():
    while True:
        msg = input("> ")
        if msg.startswith("/") or msg in ("exit", "reset", "debug"):
            if "exit" in msg:
                return
            if "reset" in msg:
                tuple(core(reset=True))
            elif "debug" in msg:
                text    = llm.detokenize(system_prompt + history)
                chainer = Utf8Chainer()
                print(chainer(text))
            else:
                print(f"Unknown command: {msg}")
        else:
            try:
                for answer in core(msg):
                    print(answer, end="", flush=True)
            except KeyboardInterrupt: pass

🟧 Да — именно так это и работает.

Когда ты максимально жёстко вычищаешь GNU‑инструменты и заставляешь всю цепочку собираться **только на LLVM‑инструментах**, ты одновременно:
- усложняешь себе жизнь на этапе сборки,
- но резко повышаешь стабильность и предсказуемость результата на этапе выполнения.

Это не интуитивно, но технически абсолютно логично.

✔ 1. Почему сборка становится сложнее

LLVM‑toolchain требует полной согласованности:
- один и тот же **C++ ABI** (libc++ или libstdc++)
- один и тот же **линковщик** (lld)
- один и тот же **RTTI/EH режим**  
- один и тот же набор **backend‑ов**  
- один и тот же **MLIR**  
- один и тот же **TableGen**  

GNU‑инструменты более «прощают» несовместимости, а LLVM‑инструменты — нет.
Поэтому:
- GNU‑линкер может собрать бинарник, который потом упадёт в runtime.
- lld откажется собирать бинарник, если ABI несовместим.

То есть LLVM‑toolchain заставляет тебя **довести систему до идеального состояния**, иначе он просто не даст собрать.

✔ 2. Почему runtime становится стабильнее

Когда всё собрано на одном ABI, одним линковщиком, одним набором библиотек:
- исчезают конфликты `std::__1` vs `std::__cxx11`
- исчезают проблемы с RTTI между разными компиляторами
- исчезают проблемы с исключениями между разными рантаймами
- исчезают проблемы с несовместимыми версиями MLIR/LLVM
- исчезают проблемы с пропавшими weak‑символами
- исчезают проблемы с разными реализациями `new/delete`
- исчезают проблемы с разными версиями `libunwind`

Это превращает runtime из «лотереи» в **детерминированную систему**.

✔ 3. Почему это особенно важно для Triton‑Ascend

Triton — это:
- MLIR
- LLVM
- Clang
- Python‑модули
- JIT‑компиляция
- динамическая линковка
- генерация IR на лету

Если ABI несовместим — всё ломается в самых неожиданных местах.

Когда ты всё собираешь на одном LLVM‑toolchain:
- JIT работает стабильно
- MLIR‑диалекты не конфликтуют
- AscendNPU‑IR не ломает RTTI
- Python‑модуль не теряет символы
- линковка не пропускает ошибки
- runtime не падает

✔ 4. Итог

Да — ты сознательно выбрал **более сложный путь сборки**, но:
- он даёт **максимальную стабильность**,
- минимизирует runtime‑ошибки,
- устраняет скрытые ABI‑конфликты,
- делает систему предсказуемой,
- и полностью исключает «фейерверки символов» при запуске.

Это ровно тот подход, который используют разработчики LLVM, MLIR, Triton и больших компиляторов.
